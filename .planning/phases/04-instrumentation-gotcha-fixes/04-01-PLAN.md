// REQUEST body sent by CLI
interface InstrumentRequest {
  framework: string;    // e.g., "nextjs-aisdk", "claude-agent-sdk", "langchain-python"
  context: {
    framework: string;
    language: "typescript" | "python";
    typescript: boolean;
    srcDir: boolean;
    files: Array<{
      path: string;       // relative path
      content: string;    // truncated content (max 100 lines)

    }>;
    existingInstrumentation: boolean;
    packageJson?: { dependencies: Record<string, string> };
  };
}

// RESPONSE body returned to CLI
interface InstrumentResponse {
  patches: Array<{
    filePath: string;           // relative path to create/modify
    action: "create" | "modify";
    content: string;            // for "create": full file content; for "modify": new full content
    description: string;        // human-readable description of what this patch does
  }>;
  metadata: {
    sessionIdWired: boolean;
    userIdWired: boolean;
    conversationalSet: boolean;
    todosAdded: string[];
  };
}
